\subsection{Phản hồi ban đầu từ người dùng tiềm năng (Q\&A thể hiện)}

Ở giai đoạn kiểm chứng, nhóm không suy diễn phản hồi tích cực khi chưa có dữ liệu
thực tế. Bộ câu hỏi dưới đây được dùng để phỏng vấn tối thiểu 10 trung tâm hoặc
đơn vị bán khóa học, sau đó cập nhật kết quả bằng số người trả lời, trích dẫn đã
được đồng ý và bằng chứng đặt cọc hoặc mua gói.

\subsubsection*{Phần 1: Hiện trạng vận hành và mức độ phù hợp ICP}

\begin{enumerate}
    \item \textbf{Đơn vị hiện bán loại khóa học nào và người học thường đăng ký
    bằng cách nào?} \\
    \textit{(Có thể chọn nhiều đáp án)}
    \begin{itemize}
        \item[$\square$] Tiếng Anh.
        \item[$\square$] Tiếng Hàn.
        \item[$\square$] Ngôn ngữ khác: \dotfill
        \item[$\square$] Luyện thi, kỹ năng hoặc chương trình đào tạo khác:
        \dotfill
        \item[$\square$] Inbox/để lại thông tin.
        \item[$\square$] Đăng ký học thử/đặt lịch tư vấn.
        \item[$\square$] Thanh toán trực tiếp.
    \end{itemize}

    \item \textbf{Trong một tuần điển hình, đơn vị livestream bao nhiêu buổi,
    tổng cộng bao nhiêu giờ và trên kênh nào?} \\
    Facebook: \dotfill \quad TikTok: \dotfill \quad Kênh khác: \dotfill

    \item \textbf{Có bao nhiêu người tham gia chuẩn bị nội dung, đứng live, trực
    bình luận và tư vấn lead? Những vai trò nào thường bị quá tải?} \\
    \textbf{Phản hồi:} \rule{0.72\linewidth}{0.4pt}

    \item \textbf{Ba khó khăn lớn nhất khi muốn tăng thêm 10--30 giờ-kênh mỗi
    tuần là gì?} \\
    \textit{(Chọn tối đa ba đáp án)}
    \begin{itemize}
        \item[$\square$] Thiếu host hoặc khó xếp lịch nhân sự.
        \item[$\square$] Thiếu ý tưởng/kịch bản và tài sản video.
        \item[$\square$] Chi phí quay dựng, host hoặc trực phiên cao.
        \item[$\square$] Nhân viên phải trả lời FAQ nhiều lần.
        \item[$\square$] Khó theo dõi lead và hiệu quả sau phiên.
        \item[$\square$] Lo ngại tài khoản, bản quyền hoặc chính sách nền tảng.
        \item[$\square$] Khác: \dotfill
    \end{itemize}

    \item \textbf{Đơn vị hiện chi bao nhiêu cho một video livestream hoặc một
    giờ vận hành, bao gồm host, quay dựng, nhân viên trực và công cụ?} \\
    \textit{Nếu chưa tách được chi phí, vui lòng ghi tổng chi phí ước tính mỗi
    tháng và số giờ phát tương ứng.} \\
    \textbf{Phản hồi:} \rule{0.72\linewidth}{0.4pt}
\end{enumerate}

\subsubsection*{Phần 2: Đánh giá mẫu video và quy trình dịch vụ}

\begin{enumerate}
    \setcounter{enumi}{5}
    \item \textbf{Sau khi xem mẫu video AI, mức độ phù hợp với thương hiệu và
    nội dung khóa học là bao nhiêu?} \\
    \textit{Chấm từ 1 (hoàn toàn không phù hợp) đến 5 (rất phù hợp):}
    \quad 1 \quad 2 \quad 3 \quad 4 \quad 5 \\
    \textbf{Lý do và thay đổi cần thiết:} \rule{0.55\linewidth}{0.4pt}

    \item \textbf{Đơn vị có sẵn những dữ liệu nào để triển khai?} \\
    \textit{(Có thể chọn nhiều đáp án)}
    \begin{itemize}
        \item[$\square$] Học phí và ưu đãi.
        \item[$\square$] Lịch khai giảng/lịch học.
        \item[$\square$] Lộ trình, giáo trình và trình độ đầu vào.
        \item[$\square$] FAQ và kịch bản tư vấn.
        \item[$\square$] Hình ảnh/video giáo viên hoặc lớp học.
        \item[$\square$] Tài liệu xác nhận quyền sử dụng hình ảnh/giọng nói.
    \end{itemize}

    \item \textbf{AI Lead Assistant nên được phép tự trả lời nội dung nào và
    bắt buộc chuyển nhân viên ở tình huống nào?} \\
    \textit{Gợi ý: học phí, lịch học, đầu vào, ưu đãi, cam kết đầu ra, khiếu nại,
    tư vấn trình độ hoặc thanh toán.} \\
    \textbf{Phản hồi:} \rule{0.72\linewidth}{0.4pt} \\
    \rule{\linewidth}{0.4pt}

    \item \textbf{Trong bốn đầu ra sau, đâu là giá trị quan trọng nhất?} \\
    \textit{(Xếp hạng từ 1 đến 4)}
    \begin{itemize}
        \item[$\square$] Có video sẵn sàng phát nhanh hơn.
        \item[$\square$] Có thêm giờ hiện diện mà không cần host liên tục.
        \item[$\square$] Trả lời FAQ và chuyển lead có cấu trúc.
        \item[$\square$] Có báo cáo giờ phát, câu hỏi và lead sau đợt.
    \end{itemize}
\end{enumerate}

\subsubsection*{Phần 3: Mức sẵn sàng trả phí và ý định mua lại}

\begin{enumerate}
    \setcounter{enumi}{10}
    \item \textbf{Nếu gói gồm một video đã duyệt, 30 giờ-kênh, FAQ cơ bản, thu
    và chuyển lead cùng báo cáo, đơn vị có sẵn sàng thử trả phí hoặc đặt cọc
    không?}
    \begin{itemize}
        \item[$\square$] Có thể mua/đặt cọc trong 30 ngày tới.
        \item[$\square$] Có quan tâm nhưng cần xem pilot hoặc báo giá chi tiết.
        \item[$\square$] Chưa sẵn sàng; lý do: \dotfill
    \end{itemize}

    \item \textbf{Khoảng ngân sách chấp nhận được cho gói trên là bao nhiêu và
    điều kiện nào phải đạt để đơn vị thanh toán?} \\
    \textbf{Ngân sách:} \dotfill \quad
    \textbf{Điều kiện:} \rule{0.43\linewidth}{0.4pt}

    \item \textbf{Sau khi dùng hết gói đầu tiên, yếu tố nào quyết định việc mua
    gói thứ hai?} \\
    \textit{(Chọn tối đa ba đáp án)}
    \begin{itemize}
        \item[$\square$] Chất lượng video và mức độ phù hợp thương hiệu.
        \item[$\square$] Tỷ lệ phát ổn định và hỗ trợ sự cố.
        \item[$\square$] Số lead hoặc lịch tư vấn thu được.
        \item[$\square$] Khối lượng công việc nhân viên được giảm.
        \item[$\square$] Chi phí trên mỗi giờ-kênh.
        \item[$\square$] Báo cáo rõ ràng và quy trình mua lại nhanh.
        \item[$\square$] Khác: \dotfill
    \end{itemize}

    \item \textbf{Đơn vị có đồng ý tham gia paid pilot và cho phép nhóm sử dụng
    dữ liệu tổng hợp, ẩn danh để đánh giá sản phẩm không?}
    \begin{itemize}
        \item[$\square$] Đồng ý tham gia và được liên hệ lại.
        \item[$\square$] Cần trao đổi thêm về phạm vi dữ liệu và bảo mật.
        \item[$\square$] Không đồng ý.
    \end{itemize}
\end{enumerate}

Khi tổng hợp khảo sát, nhóm cần báo cáo riêng số người phù hợp ICP, số người xem
mẫu, số người chấp nhận mức giá, số người đặt cọc và số người mua lại. Ý định nói
miệng không được xem tương đương với bằng chứng thanh toán.
